> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax.
> **Read first:** `DEVELOPMENT.md` (build layout + test layers), the `vmarkd-renderer-theming` skill, and `tasks/104-d2-diagram-renderer.md` (the full decision trail + Phase-0 measurements).
> **Run all e2e/vscode tests headless:** `xvfb-run -a …` (see DEVELOPMENT.md).
> **Don't commit/push** unless the user says so (project rule).

**Goal:** Render ` ```d2 ` fenced blocks offline in the vMarkd webview, faithfully for plain diagrams (nodes/edges/containers/styles/vars/globs/classes), with a LOUD raw-text fallback for the bespoke-layout shapes we don't support.

**Architecture (the SPLIT, validated in Phase 0):** A vendored **compile-only D2 Go→WASM** turns `d2` source into a semantic **graph JSON** using D2's REAL compiler (so semantics are faithful by construction). The **JS side** sizes labels with Canvas `measureText`, lays out with **@dagrejs/dagre** (compound graph for containers), and emits **SVG** themed via `currentColor`. Bespoke-layout shapes (`sequence_diagram`, `grid`, `near`, `sql_table`, `class`) are DETECTED from the graph and fall back to LOUD raw text — this is what keeps us faithful-by-construction (never a silent misrender). Sequence diagrams stay Mermaid permanently.

**What "faithful" means (the SPLIT's promise):** semantic faithfulness — correct graph JSON + container nesting — is guaranteed by the REAL compiler in WASM; geometric fidelity (edge routing, label sizing) is JS-rendered and *visually acceptable* per Phase 0 (<1% label drift with the font bundled, ZERO dagre reshuffle), **not pixel-perfect**.

**WASM output scope:** the WASM emits ONLY what the JS side needs — faithful-rendering detection (`isSequence`/`isGrid`/`nearKey`) + dagre layout (shapes, edges, labels, colours, containers). Bespoke-shape *internals* (SQL columns, class fields) are NOT emitted in Phase 1; adding them (Task 9) requires a WASM rebuild.

**Tech Stack:** Go 1.25 (build-time only, vendored prebuilt like Lute), `@terrastruct/d2` source @ commit `2446e24` (npm 0.1.33), `@dagrejs/dagre` 3.0.0, Source Sans 3 (SIL OFL, bundled), TypeScript/esbuild webview, vitest + Playwright (harness + real-VS-Code).

**Phase-0 measured (green-light):** compile-only nofonts WASM = 10.6 MB raw / **1.82 MB brotli**; cold init ~470 ms; resident ~28 MB; font-drift benign (+9% no-font, <1% with Source Sans 3), ZERO dagre reshuffle.

```go
//go:build js && wasm

package main

import (
	"encoding/json"
	"strings"
	"syscall/js"

	"oss.terrastruct.com/d2/d2compiler"
	"oss.terrastruct.com/d2/d2graph"
)

type outShape struct {
	ID        string  `json:"id"`
	IDVal     string  `json:"idVal"`
	Label     string  `json:"label"`
	Shape     string  `json:"shape"`
	Container string  `json:"container,omitempty"`
	Fill      string  `json:"fill,omitempty"`
	Stroke    string  `json:"stroke,omitempty"`
	Special   special `json:"special"`
}

type special struct {
	IsSequence  bool   `json:"isSequence"`
	IsGrid      bool   `json:"isGrid"`
	GridRows    string `json:"gridRows,omitempty"`
	GridColumns string `json:"gridColumns,omitempty"`
	NearKey     string `json:"nearKey,omitempty"`
}

type outEdge struct {
	Src      string `json:"src"`
	Dst      string `json:"dst"`
	Label    string `json:"label,omitempty"`
	SrcArrow bool   `json:"srcArrow"`
	DstArrow bool   `json:"dstArrow"`
}

type outGraph struct {
	Shapes []outShape `json:"shapes"`
	Edges  []outEdge  `json:"edges"`
}

func styleVal(s *d2graph.Scalar) string {
	if s == nil {
		return ""
	}
	return s.Value
}

func compileToJSON(src string) (string, error) {
	g, _, err := d2compiler.Compile("index", strings.NewReader(src), &d2compiler.CompileOptions{})
	if err != nil {
		return "", err
	}
	og := outGraph{}
	for _, o := range g.Objects {
		container := ""
		if o.Parent != nil && o.Parent.ID != "" {
			container = o.Parent.AbsID()
		}
		sp := special{
			IsSequence:  o.IsSequenceDiagram(),
			IsGrid:      o.IsGridDiagram(),
			GridRows:    styleVal(o.GridRows),
			GridColumns: styleVal(o.GridColumns),
		}
		if o.NearKey != nil {
			sp.NearKey = strings.Join(d2graph.Key(o.NearKey), ".")
		}
		og.Shapes = append(og.Shapes, outShape{
			ID:        o.AbsID(),
			IDVal:     o.IDVal,
			Label:     o.Label.Value,
			Shape:     o.Shape.Value,
			Container: container,
			Fill:      styleVal(o.Style.Fill),
			Stroke:    styleVal(o.Style.Stroke),
			Special:   sp,
		})
	}
	for _, e := range g.Edges {
		var src, dst, label string
		if e.Src != nil {
			src = e.Src.AbsID()
		}
		if e.Dst != nil {
			dst = e.Dst.AbsID()
		}
		label = e.Label.Value
		og.Edges = append(og.Edges, outEdge{
			Src: src, Dst: dst, Label: label,
			SrcArrow: e.SrcArrow, DstArrow: e.DstArrow,
		})
	}
	b, err := json.Marshal(og)
	if err != nil {
		return "", err
	}
	return string(b), nil
}

func d2compile(this js.Value, args []js.Value) interface{} {
	if len(args) < 1 {
		return map[string]interface{}{"error": "missing d2 source"}
	}
	out, err := compileToJSON(args[0].String())
	if err != nil {
		return map[string]interface{}{"error": err.Error()}
	}
	return map[string]interface{}{"graph": out}
}

func main() {
	js.Global().Set("d2compile", js.FuncOf(d2compile))
	select {}
}
```

> **Contract note:** `outShape`/`outEdge`/`outGraph` are the Go mirror of the **`D2Graph` TypeScript interface in `d2-wasm.ts` (Task 3) — that interface is the source of truth.** Any field change must update BOTH sides and is caught by `d2-wasm.test.ts`.

```js
async function syncD2() {
  const vendorDir = path.resolve('media-src/vendor/d2')
  const targetDir = path.resolve('media/vditor/dist/js/d2')
  let source
  try {
    source = JSON.parse(await fs.readFile(path.join(vendorDir, 'source.json'), 'utf8'))
  } catch {
    return
  }
  for (const [name, meta] of Object.entries(source.files)) {
    const buf = await fs.readFile(path.join(vendorDir, name))
    const got = createHash('sha256').update(buf).digest('hex')
    if (got !== meta.sha256) {
      throw new Error(`[d2] vendored ${name} sha256 mismatch:\n  expected ${meta.sha256}\n  got      ${got}`)
    }
  }
  await fs.mkdir(targetDir, { recursive: true })
  await fs.copyFile(path.join(vendorDir, 'd2-compile.wasm'), path.join(targetDir, 'd2-compile.wasm'))
  await fs.copyFile(path.join(vendorDir, 'wasm_exec.js'), path.join(targetDir, 'wasm_exec.js'))
  console.log(`[d2] vendored v${source.version} verified + installed`)
}
```

## Task 3: WASM loader — `media-src/src/d2-wasm.ts`

**Files:** Create `media-src/src/d2-wasm.ts`. Test: `media-src/src/d2-wasm.test.ts` (vitest, node).

The boot sequence (Phase-0-verified): load `wasm_exec.js` (defines `globalThis.Go`), instantiate the wasm, call `go.run(instance)` WITHOUT awaiting (it blocks on `select{}`), wait one tick, then `window.d2compile(src)` returns `{graph: jsonString}` or `{error}` synchronously. **Two layers:** `window.d2compile` is *synchronous* once the WASM is booted; the exported `compileD2(cdn, src)` is *async* because it first boots the WASM, and on any failure it **resolves** (never rejects) with `{error}`.

- [ ] **Step 3: Write the loader** `d2-wasm.ts` (browser; lazy, cached). `addScript` is reused from custom-diagrams (or inline a script-tag loader); the wasm is fetched same-origin from the cdn base. Both URLs carry a `?v=` cache-buster (= `source.json` version) so a cached webview never serves stale bytes across an extension update — the same rule Vditor's mermaid/echarts assets follow:

```ts
// Lazy loader for the vendored compile-only D2 WASM. Boots Go's runtime once,
// caches the global window.d2compile, and exposes compileD2(src) -> graph object.
// CSP: wasm_exec.js uses Function() to instantiate the module, so it needs
// script-src 'unsafe-eval' (already shipped). Per W3C, 'unsafe-eval' also authorizes
// WebAssembly.instantiate — but VERIFY in the real webview and add 'wasm-unsafe-eval'
// to html-builder.ts if the WASM fails to boot (the vmarkd-renderer-theming skill flags this).
declare const window: any

export interface D2Graph {
  shapes: Array<{
    id: string; idVal: string; label: string; shape: string; container?: string
    fill?: string; stroke?: string
    special: { isSequence: boolean; isGrid: boolean; gridRows?: string; gridColumns?: string; nearKey?: string }
  }>
  edges: Array<{ src: string; dst: string; label?: string; srcArrow: boolean; dstArrow: boolean }>
}

// Cache-buster: MUST equal media-src/vendor/d2/source.json "version" (bump both on a D2 update).
const D2_VER = '0.1.33'

let bootPromise: Promise<((src: string) => any) | null> | null = null

function loadScript(src: string, id: string): Promise<void> {
  return new Promise((resolve) => {
    if (document.getElementById(id)) return resolve()
    const s = document.createElement('script')
    s.id = id; s.src = src
    s.onload = () => resolve()
    s.onerror = () => resolve()
    document.head.appendChild(s)
  })
}

export function bootD2(cdn: string): Promise<((src: string) => any) | null> {
  if (bootPromise) return bootPromise
  bootPromise = (async () => {
    await loadScript(`${cdn}/dist/js/d2/wasm_exec.js?v=${D2_VER}`, 'vditorD2WasmExec')
    if (!window.Go) return null
    let buf: ArrayBuffer
    try {
      const resp = await fetch(`${cdn}/dist/js/d2/d2-compile.wasm?v=${D2_VER}`)
      buf = await resp.arrayBuffer()
    } catch { return null }
    const go = new window.Go()
    const { instance } = await WebAssembly.instantiate(buf, go.importObject)
    go.run(instance) // blocks on select{}; do not await
    // Phase 0: cold init ~470 ms; d2compile registers within a few frames after go.run().
    // 50 rAF (~0.8 s @60 fps) is a generous safety margin, NOT a tuned constant. If the
    // global never registers we return null -> compileD2 -> {error:'d2 wasm unavailable'},
    // which renderD2 logs distinctly from a compile error (see D2-1).
    for (let i = 0; i < 50 && typeof window.d2compile !== 'function'; i++) {
      await new Promise((r) => requestAnimationFrame(r))
    }
    return typeof window.d2compile === 'function' ? window.d2compile : null
  })()
  return bootPromise
}

export async function compileD2(cdn: string, src: string): Promise<D2Graph | { error: string }> {
  const fn = await bootD2(cdn)
  if (!fn) return { error: 'd2 wasm unavailable' }
  const out = fn(src)
  if (out.error) return { error: out.error }
  return JSON.parse(out.graph) as D2Graph
}
```

- [ ] **Step 3: Write `d2-render.ts`** — port `mvp-render.mjs` to TS, taking an injectable `measure` fn, and replace the hand-tuned `shapePadding` with the **faithful `dimsToFit`** (constants below from D2 `lib/shape`; `content = label + INNER_LABEL_PADDING(5)`):

```ts
import dagre from '@dagrejs/dagre'
import type { D2Graph } from './d2-wasm'

const FONT_SIZE = 16, EDGE_FONT_SIZE = 14, INNER_PAD = 5, P = 40 // d2 defaultPadding
export type Sizer = (text: string, fontSize?: number) => { w: number; h: number } // import from d2-render.ts to type a custom measure fn

const ceil = Math.ceil, SQRT2 = Math.SQRT2

// Faithful D2 lib/shape GetDimensionsToFit. (w,h) = labelDims + INNER_PAD already applied.
function dimsToFit(shape: string, w: number, h: number): { w: number; h: number } {
  switch (shape) {
    case 'square':      { const s = ceil(Math.max(w + P, h + P)); return { w: s, h: s } }
    case 'circle':      { const d = ceil(SQRT2 * Math.max(w + P / SQRT2, h + P / SQRT2)); return { w: d, h: d } }
    case 'oval':        { const t = Math.atan2(h, w); return { w: ceil(SQRT2 * (w + P * Math.cos(t))), h: ceil(SQRT2 * (h + P * Math.sin(t))) } }
    case 'diamond':     return { w: ceil(2 * (w + 10)), h: ceil(2 * (h + 20)) }
    case 'hexagon':     return { w: ceil(1.5 * (w + 20)), h: ceil(1.5 * (h + 20)) }
    case 'cylinder':    return { w: ceil(w + P), h: ceil(h + 20 + 72) }
    case 'parallelogram': return { w: ceil(w + P + 52), h: ceil(h + P) }
    case 'document':    return { w: ceil(w + P), h: ceil((h + 29.59) * 18.925 / 14) }
    case 'page':        return { w: ceil(w + P), h: ceil(h + 60.348) }
    default:            return { w: ceil(w + P), h: ceil(h + P) } // rectangle/"": (40,40)
  }
}

function shapeBox(shape: string, m: { w: number; h: number }) {
  return dimsToFit(shape, m.w + INNER_PAD, m.h + INNER_PAD)
}

export function renderD2Graph(graph: D2Graph, measure: Sizer): string {
  const g: any = new (dagre as any).graphlib.Graph({ compound: true, multigraph: true })
  // d2's dagre tuning (lib/d2dagrelayout): nodesep 60, ranksep max(100,…), edgesep 20
  g.setGraph({ rankdir: 'TB', nodesep: 60, ranksep: 100, edgesep: 20, marginx: 10, marginy: 10 })
  g.setDefaultEdgeLabel(() => ({}))

  const containers = new Set<string>()
  for (const s of graph.shapes) if (s.container) containers.add(s.container)

  for (const s of graph.shapes) {
    const isContainer = containers.has(s.id)
    const lbl = measure(s.label)
    if (isContainer) {
      g.setNode(s.id, { label: s.label, isContainer: true, shape: s.shape, fill: s.fill, stroke: s.stroke, headerH: lbl.h + 10 })
    } else {
      const box = shapeBox(s.shape, lbl)
      g.setNode(s.id, { width: box.w, height: box.h, label: s.label, shape: s.shape, fill: s.fill, stroke: s.stroke, isContainer: false })
    }
  }
  for (const s of graph.shapes) if (s.container && g.hasNode(s.container)) g.setParent(s.id, s.container)
  let ei = 0
  for (const e of graph.edges) {
    if (!g.hasNode(e.src) || !g.hasNode(e.dst)) continue
    const el = e.label ? measure(e.label, EDGE_FONT_SIZE) : { w: 0, h: 0 }
    g.setEdge(e.src, e.dst, { label: e.label || '', width: el.w, height: el.h, srcArrow: e.srcArrow, dstArrow: e.dstArrow }, 'e' + ei++)
  }
  ;(dagre as any).layout(g)
  return toSVG(g)
}

const esc = (s: any) => String(s).replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')

function arrow(x: number, y: number, angle: number, color: string): string {
  const len = 10
  const a1 = angle + Math.PI - 0.4, a2 = angle + Math.PI + 0.4
  const x1 = x + len * Math.cos(a1), y1 = y + len * Math.sin(a1)
  const x2 = x + len * Math.cos(a2), y2 = y + len * Math.sin(a2)
  return `<polygon points="${x},${y} ${x1.toFixed(1)},${y1.toFixed(1)} ${x2.toFixed(1)},${y2.toFixed(1)}" fill="${color}"/>`
}

function toSVG(g: any): string {
  const gg = g.graph(), W = ceil(gg.width) + 20, H = ceil(gg.height) + 20, OFF = 10
  // currentColor theming (our house convention — themeSvg also re-asserts it)
  const stroke = 'currentColor'
  const parts: string[] = [`<svg xmlns="http://www.w3.org/2000/svg" width="${W}" height="${H}" viewBox="0 0 ${W} ${H}" font-family='"Source Sans 3","Source Sans Pro",system-ui,sans-serif'>`]
  for (const eo of g.edges()) {
    const e = g.edge(eo)
    const pts = e.points.map((p: any) => [p.x + OFF, p.y + OFF])
    const d = pts.map((p: number[], i: number) => (i === 0 ? 'M' : 'L') + p[0].toFixed(1) + ',' + p[1].toFixed(1)).join(' ')
    parts.push(`<path d="${d}" fill="none" stroke="${stroke}" stroke-width="2"/>`)
    const n = pts.length
    if (e.dstArrow !== false) { const a = Math.atan2(pts[n-1][1]-pts[n-2][1], pts[n-1][0]-pts[n-2][0]); parts.push(arrow(pts[n-1][0], pts[n-1][1], a, stroke)) }
    if (e.srcArrow === true) { const a = Math.atan2(pts[0][1]-pts[1][1], pts[0][0]-pts[1][0]); parts.push(arrow(pts[0][0], pts[0][1], a, stroke)) }
    if (e.label && e.x != null) {
      parts.push(`<text x="${(e.x+OFF).toFixed(1)}" y="${(e.y+OFF).toFixed(1)}" font-size="${EDGE_FONT_SIZE}" text-anchor="middle" dominant-baseline="middle" fill="${stroke}">${esc(e.label)}</text>`)
    }
  }
  const ids = g.nodes().slice().sort((a: string, b: string) => (g.node(a).isContainer ? 0 : 1) - (g.node(b).isContainer ? 0 : 1))
  for (const id of ids) {
    const n = g.node(id); const x = n.x + OFF, y = n.y + OFF, w = n.width, h = n.height, left = x - w/2, top = y - h/2
    const fill = n.fill || 'transparent'      // explicit style.fill kept; else transparent (theme bg shows)
    if (n.isContainer) {
      parts.push(`<rect x="${left.toFixed(1)}" y="${top.toFixed(1)}" width="${w.toFixed(1)}" height="${h.toFixed(1)}" rx="6" fill="${fill}" stroke="${stroke}" stroke-width="2" fill-opacity="${n.fill?'1':'0.04'}"/>`)
      parts.push(`<text x="${(left+8).toFixed(1)}" y="${(top-6).toFixed(1)}" font-size="${FONT_SIZE}" fill="${stroke}">${esc(n.label)}</text>`)
      continue
    }
    switch (n.shape) {
      case 'circle': case 'oval':
        parts.push(`<ellipse cx="${x.toFixed(1)}" cy="${y.toFixed(1)}" rx="${(w/2).toFixed(1)}" ry="${(h/2).toFixed(1)}" fill="${fill}" stroke="${stroke}" stroke-width="2"/>`); break
      case 'diamond':
        parts.push(`<polygon points="${x},${top.toFixed(1)} ${(left+w).toFixed(1)},${y} ${x},${(top+h).toFixed(1)} ${left.toFixed(1)},${y}" fill="${fill}" stroke="${stroke}" stroke-width="2"/>`); break
      case 'hexagon': { const i = w*0.22
        parts.push(`<polygon points="${(left+i).toFixed(1)},${top.toFixed(1)} ${(left+w-i).toFixed(1)},${top.toFixed(1)} ${(left+w).toFixed(1)},${y} ${(left+w-i).toFixed(1)},${(top+h).toFixed(1)} ${(left+i).toFixed(1)},${(top+h).toFixed(1)} ${left.toFixed(1)},${y}" fill="${fill}" stroke="${stroke}" stroke-width="2"/>`); break }
      case 'cylinder': { const ry = Math.min(h*0.12, 12)
        parts.push(`<path d="M${left},${(top+ry).toFixed(1)} A${(w/2).toFixed(1)},${ry.toFixed(1)} 0 0 1 ${(left+w).toFixed(1)},${(top+ry).toFixed(1)} L${(left+w).toFixed(1)},${(top+h-ry).toFixed(1)} A${(w/2).toFixed(1)},${ry.toFixed(1)} 0 0 1 ${left},${(top+h-ry).toFixed(1)} Z" fill="${fill}" stroke="${stroke}" stroke-width="2"/>`)
        parts.push(`<path d="M${left},${(top+ry).toFixed(1)} A${(w/2).toFixed(1)},${ry.toFixed(1)} 0 0 0 ${(left+w).toFixed(1)},${(top+ry).toFixed(1)}" fill="none" stroke="${stroke}" stroke-width="2"/>`); break }
      default:
        parts.push(`<rect x="${left.toFixed(1)}" y="${top.toFixed(1)}" width="${w.toFixed(1)}" height="${h.toFixed(1)}" rx="4" fill="${fill}" stroke="${stroke}" stroke-width="2"/>`)
    }
    parts.push(`<text x="${x.toFixed(1)}" y="${y.toFixed(1)}" font-size="${FONT_SIZE}" text-anchor="middle" dominant-baseline="central" fill="${stroke}">${esc(n.label)}</text>`)
  }
  parts.push('</svg>')
  return parts.join('\n')
}
```

// CRITICAL — faithful-by-construction (NON-NEGOTIABLE): returns a reason string if the
// graph uses a bespoke layout/shape dagre cannot faithfully render (sequence_diagram,
// grid, near, sql_table, class) => the caller MUST show raw text LOUDLY, never a wrong
// picture. Never remove or weaken this without a full faithfulness audit. Single
// enforcement point: custom-diagrams.ts renderD2.
export function unsupportedReason(graph: D2Graph): string | null {
  for (const s of graph.shapes) {
    if (s.special.isSequence) return 'sequence_diagram (use ```mermaid)'
    if (s.special.isGrid) return 'grid layout'
    if (s.special.nearKey) return 'near positioning'
    if (s.shape === 'sql_table' || s.shape === 'class') return `${s.shape} shape`
  }
  return null
}

```ts
import { compileD2 } from './d2-wasm'
import { renderD2Graph, canvasMeasure, unsupportedReason } from './d2-render'

export function renderD2(root?: ParentNode): void {
  const container = root ?? document
  // findBlocks already skips IR/WYSIWYG edit-surface markers (.vditor-ir__marker--pre,
  // .vditor-wysiwyg__pre) and already-[data-processed] blocks — D2 inherits that guard.
  const blocks = findBlocks(container, 'd2')
  if (!blocks.length) return
  const cdn = getCdn()
  for (const { wrapper, code } of blocks) {
    // D2 is ASYNC (WASM boot + compile), unlike the synchronous renderers. Mark the block
    // processed UP-FRONT so a re-firing observer can't double-render it while compileD2 is
    // pending (synchronous renderers can set this at the end; D2 cannot).
    wrapper.setAttribute('data-processed', 'true')
    compileD2(cdn, code).then((res) => {
      if ((res as any).error) {
        const err = (res as any).error
        // D2-1: distinguish a boot/timeout ('d2 wasm unavailable') from a real compile error
        // via the existing webview→host log helper, so a stuck WASM isn't mistaken for bad
        // d2 syntax. Both leave the source visible (loud); neither throws.
        // logToHost(err === 'd2 wasm unavailable' ? 'd2: WASM boot failed' : `d2: compile error: ${err}`)
        return
      }
      const graph = res as any
      const reason = unsupportedReason(graph)
      if (reason) {
        // LOUD fallback (faithful-by-construction, NON-NEGOTIABLE): raw source + a note,
        // NEVER a partial/wrong picture. This is the single enforcement point for unsupportedReason.
        wrapper.innerHTML = ''
        const pre = document.createElement('pre')
        pre.className = 'language-d2-unsupported'
        pre.textContent = code
        const note = document.createElement('div')
        note.className = 'd2-unsupported-note'
        note.textContent = `d2: ${reason} not supported — showing source`
        wrapper.append(note, pre)
        return
      }
      wrapper.innerHTML = renderD2Graph(graph, canvasMeasure)
      const svg = wrapper.querySelector('svg')
      if (svg) themeSvg(svg)
    }).catch(() => { /* leave source visible */ })
  }
}

export function reRenderD2(root?: ParentNode): void {
  const container = root ?? document
  for (const pane of Array.from(container.querySelectorAll<HTMLElement>(PANE_SEL))) {
    for (const el of Array.from(pane.querySelectorAll<HTMLElement>(
      'code.language-d2[data-processed], div.language-d2[data-processed]'))) {
      el.removeAttribute('data-processed'); el.innerHTML = ''
    }
  }
  renderD2(container)
}
```
